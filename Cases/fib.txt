program fib;
    var i, n, primer, segundo, sig:int;

    main {
        n = 5;

        primer = 0;
        segundo = 1;
        i = 2;
        print(primer);
        print(segundo);

        do {
            sig = primer + segundo;
            print(sig);
            primer = segundo;
            segundo= sig;
            i = i + 1;
        } while(i < n);
    }
end
